- dotnet restore (whole solution, not just src/)
- dotnet build --no-restore --configuration Release

- needs: build
- strategy.matrix.module: [Users, Products, Orders]
- strategy.fail-fast: false
- actions/checkout@v4
- actions/setup-dotnet@v4 with dotnet-version: '10.0.x'
- dotnet restore tests/WorktreeApi.Tests.fsproj
- dotnet run --project tests/WorktreeApi.Tests.fsproj -- --filter-test-list "${{ matrix.module }}"

**Stage 3 - cleanup job:**
- runs-on: ubuntu-latest
- needs: [test]
- if: always()
- actions/checkout@v4
- git worktree list, git worktree prune -v, git worktree list

IMPORTANT: Use `--filter-test-list` NOT `--filter`. Use `dotnet-version: '10.0.x'` NOT `9.0.x`.
